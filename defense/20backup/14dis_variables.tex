% --+ 12.14 DIS VARIABLES +-----------------------------------------------------
\begin{frame}{DIS Variables}
    \label{12.14::dis_variables}

    \vspace{24pt}

    If \ef{$E_b$} is the beam energy, \ef{$E'$} and \ef{$\theta_C$} the energy and angle of the scattered $e^-$, and \ef{$E_h$}, \ef{$p_h$}, and \ef{$M_h$} are the scattered hadron's energy, momentum, and mass, then:

    \vspace{12pt}

    \begin{itemize}
        \item
            Transferred momentum: \ef{$Q^2 \labeq 4E_bE'\sin^2(\theta_C/2)$}.

        \item
            Virtual photon energy: \ef{$\nu \labeq E_b - E'$}.

        \item
            Hadron energy fraction: \ef{$z_h \labeq E_h/\nu$}.

        \item
            Hadron's transverse momentum: \ef{$p_T^2 = p_h^2(1 - \cos^2(\theta_{PQ}))$}.

        \item
            Angle between the leptonic and hadronic planes: \ef{$\phi_{PQ}$}.

        \item
            Squared mass of the final hadronic state: \ef{$W^2 = M_h^2 + 2M_h\nu - Q^2$}.

        \item
            Bjorken-Y: \ef{$Y_b \labeq \nu/E_b$}.
    \end{itemize}

    \backref{11.22::dis_cuts}
\end{frame}
