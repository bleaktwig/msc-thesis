% --+ 10.21 RG-E Target +-------------------------------------------------------
\begin{frame}{Run Group E (RG-E) Target}
    \label{10.21::rge_target}

    \begin{itemize}
        \item
            The RG-E target was designed to enhance our understanding of hadronisation in the nuclear medium, it's main observable being \ef{hadronic multiplicity ratio}.

        \item
            To cancel time-dependent systematic effects, the system contains two objectives: a \ef{deuterium} and a \ef{solid} target.

        \item
            The experiment is set to run at \ef{Hall B} of \ef{Jefferson Lab (JLab)}.
    \end{itemize}

    \end{center}
\end{frame}
